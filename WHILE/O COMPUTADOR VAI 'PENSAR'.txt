# MELHORE O JOGO DO DESAFIO 028 ONDE O COMPUTADOR VAI 'PENSAR' EM UM NUMERO ENTRE 0 E 10. SÓ QUE AGORA O JOGADOR VAI TENTAR
# ADIVINHAR ATE ACERTAR, MOSTRANDO NO FINAL QUANTOS PALPITES FORAM NECESSARIOS PARA VENCER.

from random import randint
computador = randint(0, 10)
print('Sou seu COMPUTADOR... Acbei de pensar em um número entre 0 e 10.')
print('Será que você consegue adivinhar qual foi?')
acertou = False
palpites = 0
while not acertou:
    jogador = int(input('Qual é seu palpite?'))
    palpites += 1
    if jogador == computador:
        acertou = True
    else:
        if jogador < computador:
            print('MAIS... Tente mais uma vez.')
        elif jogador > computador:
            print('MENOS... Tente mais uma vez.')
print('Você acertou com {} tentativas. Prabens!'.format(palpites))
