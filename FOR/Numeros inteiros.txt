# DESENVOLVA UM PROGRAMA QUE LEIA 6 NUMEROS INTEIROS E MOSTRE A SOMA APENAS DAQUELES QUE FORAM PARES. SE O VALOR DIGITADO FOR IMPAR, DESCONSIDERE-O.
s = 0 #acumulador
cont = 0 #contador
for c in range(1, 7):
    i = int(input('Digite um numero: '))
    if i % 2 == 0:
        s = s + i
        cont = cont + 1
print('A quantidade de numeros PARES digitados foi {} e a soma de todos esses numero é: {}'.format(cont, s))
print('fim')
